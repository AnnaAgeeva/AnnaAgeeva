#include <stdio.h>
#include <iostream>

using namespace std ;

class Lamp{ //задаем класс с названием Lamp
  public: //чтобы работать с внутренностями(открытые данные)
    int turned; //свойство класса
    Lamp(int a) { //конструктор
        turned = a;
        
    }

    int swich() { //создаем метод
      if (turned == 0){ //задаем условие "Если turned равно 0"
        cout << "Лампа выключена!" << '\n'; //тогда выводим в консоль слово "Лампа выключена"
        turned =1; // и тогда turned меняет значение с 0 на 1
        cout<<"Включаем лампу"<<"\n"; //выводим в консоль слово "Лампа включена"
      }
      if (turned == 1){ //задаем второе условие "Если turned равна 1"
         cout << "Лампа включена!" << '\n'; //тогда выводим в консоль слово "Лампа включена"
        turned = 0; //и тогда turned меняет значение с 1 на 0
        cout<<"Выключаем лампу"<<"\n"; //выводим в консоль слово "Лампа выключена"
      }
     return 0; 
    };
};



int main () { //создаем программу
  Lamp lampochka(0); //создаем объект класса Lamp с названием lampochka(начальное значение turned=0)
  lampochka.swich(); //обращаемся к объекту класса Lamp к методу swich
  return 0;
  }
